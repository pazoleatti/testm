/**
 * Обновление справочника "Классификатор расходов ПАО Сбербанк для целей налогового учёта" id = 27.
 * https://jira.aplana.com/browse/SBRFACCTAX-15298 0.8.4. Написать скрипт для обновления справочников доходов и расходов.
 */

update()

@Field
def providerCache = [:]

@Field
def refbookId = 27

void update() {
    // получить все записи из базы
    def date = Date.parse('dd.MM.yyyy', '01.01.2016')
    def filter = null
    RefBookDataProvider provider = formDataService.getRefBookProvider(refBookFactory, refbookId, providerCache)
    def records = provider.getRecords(date, null, filter, null)

    // новые версии - измененные значения
    def newVersionsMap = [:]
    // новые записи - строк файла нет в базе
    def newRecords = []
    // записи с 01.01.2016 для обновления значении
    def updateVersions = []
    // старые записи - записей бд нет в файле
    def closeRecords = []
    // ненужные версии - записей бд нет в файле (период версии с 01.01.2016)
    def deleteVersions = []
    def escape = 0

    // отобрать записи которые надо обновить
    for (def record : records) {
        // найти среди новых значении строку соответствующую текущей записи справочника
        def findValues = valuesList.find { isOneRecord(record, it) }

        // если для записи нет строки в файле, то запись больше не актуальна
        if (!findValues) {
            closeRecords.add(record)
            continue
        }

        // если значения не изменились, то пропустить
        if (equals(record, findValues)) {
            escape++
            continue
        }

        // значения изменились, то добавить новую версию записи
        newVersionsMap[record] = findValues
    }

    // отобрать строки файла, для которых нет записей в базе и добавить в базу с датой начала актуальности 01.01.2016
    for (def values : valuesList) {
        def findRecord = records.find { isOneRecord(it, values) }
        if (!findRecord) {
            newRecords.add(values)
        }
    }

    TAUserInfo taUserInfo = new TAUserInfo()
    taUserInfo.user = user
    logger.taUserInfo = taUserInfo
    def versionEnd = date - 1
	
	// среди новых версии отобрать версии 01.01.2016 и обновить их значения в базе
    if (newVersionsMap) {
        def uniqueRecordIds = newVersionsMap.keySet().toArray().collect { it?.record_id?.value }
        def recordVersionMap = provider.getRecordsVersionStart(uniqueRecordIds)
        newVersionsMap.keySet().toArray().each { record ->
            def dateStart = recordVersionMap[record?.record_id?.value]
            if (dateStart == date) {
                updateVersions.add(record)
            }
        }
        updateVersions.each { updateVersion ->
            def values = newVersionsMap[updateVersion]
            def recordMap = getRefbookValues(values)
            provider.updateRecordVersion(logger, updateVersion?.record_id?.value, null, null, recordMap)
        }
        if (updateVersions) {
            // убрать обновляемые версии 01.01.2016 из новых версии
            updateVersions.each { updateVersion ->
                newVersionsMap.remove(updateVersion)
            }
            logger.info('Версии 01.01.2016 обновлено: ' + updateVersions.size())
        }
    }

    // записать новые версии в базу
    if (newVersionsMap) {
        def newVersions = []
        newVersionsMap.each { record, values ->
            RefBookRecord refBookRecord = new RefBookRecord()
            def id = record?.record_id?.value
            def commonId = provider.getRecordId(id)
            refBookRecord.recordId = commonId
            refBookRecord.values = getRefbookValues(values)
            newVersions.add(refBookRecord)
        }
        // ограничить старые версии
        def uniqueRecordIds = newVersionsMap.keySet().toArray().collect { it?.record_id?.value }
        provider.updateRecordsVersionEnd(logger, versionEnd, uniqueRecordIds)
        // добавить новые версии
        provider.createRecordVersion(logger, date, null, newVersions)
        logger.info('Новых версии добавлено: ' + newVersions.size())
    }

    // записать новые записи в базу
    if (newRecords) {
        def insertNewRecords = []
        for (def values : newRecords) {
            def recordMap = getRefbookValues(values)
            insertNewRecords.add(recordMap)
        }
        provider.insertRecords(taUserInfo, date, insertNewRecords)
        logger.info('Новых записей добавлено: ' + insertNewRecords.size())
    }
	
	// для неактуальных записей - если запись с 01.01.2016, то удалить версию, иначе задать окончание 31.12.2015
    if (closeRecords) {
        def uniqueRecordIds = closeRecords.collect { it?.record_id?.value }
        def recordVersionMap = provider.getRecordsVersionStart(uniqueRecordIds)
        // отобрать среди неактуальных записей версии для удаления
        closeRecords.each { record ->
            def dateStart = recordVersionMap[record?.record_id?.value]
            if (dateStart == date) {
                deleteVersions.add(record)
            }
        }
        if (deleteVersions) {
            closeRecords.removeAll(deleteVersions)
            uniqueRecordIds = deleteVersions.collect { it?.record_id?.value }
            provider.deleteRecordVersions(logger, uniqueRecordIds)
            logger.info('Версии неактуальных записей 01.01.2016 удалено: ' + deleteVersions.size())
        }
    }

    // задать дату окончания актуальности 31.12.2015 для неактуальных записей
    if (closeRecords) {
        def uniqueRecordIds = closeRecords.collect { it?.record_id?.value }
        uniqueRecordIds.each {
            provider.updateRecordsVersionEnd(logger, versionEnd, [it])
        }
        // provider.updateRecordsVersionEnd(logger, versionEnd, uniqueRecordIds)
        logger.info('Дата окончания актуальности для неактуальных записей задано: ' + uniqueRecordIds.size())
    }

    if (escape) {
        logger.info('Количество неизменившихся записей: ' + escape)
    }
}

/**
 * Проверить соответствует ли записть из бд строке из файла.
 *
 * @param record запись справочника
 * @param values значения из файла
 */
def isOneRecord(def record, def values) {
    return (record?.CODE?.value == values[0] &&
            record?.BALANCE_ACCOUNT?.value == (values[3] ?: null) &&
            record?.OPU?.value == (values[4] ?: null))
}

/**
 * Сравнить значения некоторых атрибутов из базы и из файла.
 *
 * @param record запись справочника
 * @param values значения из файла
 */
def equals(def record, def values) {
    for (def alias : equalsAliases) {
        def index = allAliases.indexOf(alias)
        if (record[alias]?.value != (values[index] ?: null)) {
            return false
        }
    }
    return true
}

/**
 * Получить значения атрибутов записи/версии.
 *
 * @param values значения из файла
 */
def getRefbookValues(def values) {
    def record = [:]
    allAliases.each { alias ->
        def index = allAliases.indexOf(alias)
        record[alias] = new RefBookValue(RefBookAttributeType.STRING, values[index])
    }
    // BALANCE_ACCOUNT + OPU
    def number = (values[3] + values[4]).replace('.', '')
    record.NUMBER = new RefBookValue(RefBookAttributeType.STRING, number)
    return record
}
