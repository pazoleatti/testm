/**
# Описание:
	Добавляет пустой макет декларации

# Инструкция:
	1. Войти в приложение;
	2. Перейти на страницу, адрес которой необходимо задать в адресной строке. 
	   Адрес страницы формируется из текущего адреса посредством замены его окончания на:
		   .../gwtapp/#!scriptExecution
	   Например так должно получиться: 
		   https://172.16.127.21/taxaccounting-next2/gwtapp/#!scriptExecution;
	5. На открывшейся странице нажать кнопку "Выполнить"; 
*/
try { 
	def sqls = """
		INSERT INTO DECLARATION_TYPE (ID, TAX_TYPE, NAME, STATUS, IS_IFRS) VALUES ('22', 'I', 'Приложение №2 к декларации по налогу на прибыль (Банк)', '0', '0');
		INSERT INTO DECLARATION_TEMPLATE (ID, DECLARATION_TYPE_ID, VERSION, NAME, STATUS) VALUES (seq_declaration_template.nextval, '22', TO_DATE('2015-01-01 00:00:00', 'YYYY-MM-DD HH24:MI:SS'), 'Приложение №2 к декларации по налогу на прибыль (Банк)', '0')
	"""
	sqls.split(";").each { sql ->
		namedParameterJdbcTemplate.update(sql, [:])
	}
} catch (Exception ex) {
    logger.error("Ошибка: ${ex.getLocalizedMessage()}. Все действия, выполненные скриптом, были отменены")
}
